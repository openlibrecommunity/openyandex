Validate JSONL:
python3 - <<'PY'
import json
for p in [
    '.workflows/metadata/browser-dll-seed-pack.jsonl',
    '.workflows/metadata/active-pe-triage.jsonl',
    '.workflows/metadata/active-widget-elf-triage.jsonl',
    '.workflows/metadata/active-pdb-probes.jsonl',
]:
    rows = 0
    with open(p, encoding='utf-8') as f:
        for line in f:
            if line.strip():
                json.loads(line)
                rows += 1
    print(p, rows)
PY
